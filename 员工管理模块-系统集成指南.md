### 2. 权限集成 ✅

#### 增强的权限系统
已在 `client/src/contexts/AuthContext.jsx` 中添加完善的权限检查：

```javascript
// 角色检查
hasRole(roles) - 检查用户是否拥有指定角色

// 权限检查
hasPermission(permission) - 检查用户是否拥有指定权限

// 管理员检查
isAdmin() - 检查是否为管理员

// 员工模块访问权限
canAccessEmployeeModule() - 检查是否可以访问员工模块

// 员工管理权限
canManageEmployees() - 检查是否可以管理员工
```

#### 权限级别

**管理员 (admin)**：
- ✅ 查看所有员工
- ✅ 创建员工
- ✅ 编辑所有员工
- ✅ 删除/停用员工
- ✅ 重置密码
- ✅ 批量导入
- ✅ 角色管理
- ✅ 权限配置

**经理 (manager)**：
- ✅ 查看所有员工
- ✅ 查看员工详情
- ✅ 编辑部门员工
- ⚠️ 不能创建员工
- ⚠️ 不能删除员工
- ⚠️ 不能重置密码
- ⚠️ 不能批量导入

**普通员工 (sales/operator)**：
- ✅ 查看自己的信息
- ✅ 修改自己的信息
- ✅ 修改自己的密码
- ❌ 不能访问员工管理模块
- ❌ 不能查看其他员工

#### 页面级权限控制

使用 `PrivateRoute` 组件进行路由级别的权限控制：

```javascript
<PrivateRoute roles={['admin', 'manager']}>
  <EmployeeList />
</PrivateRoute>
```

#### 组件级权限控制

在组件内部使用 `useAuth` Hook：

```javascript
const { hasPermission, isAdmin, canManageEmployees } = useAuth();

// 条件渲染
{isAdmin() && (
  <Button onClick={handleDelete}>删除员工</Button>
)}

// 功能禁用
<Button 
  disabled={!canManageEmployees()}
  onClick={handleEdit}
>
  编辑
</Button>
```

### 3. 数据一致性处理

#### 员工状态变更的连锁反应

**场景1：员工离职**
```javascript
// 1. 修改员工状态为 'inactive'
await employeeAPI.updateEmployeeStatus(id, 'inactive');

// 2. 处理关联数据
// - 取消该员工的所有客户分配
// - 转移进行中的任务
// - 记录离职日期
// - 归档员工数据

// 3. 通知相关模块
// - 客户管理模块：重新分配客户
// - 任务管理模块：转移任务
// - 权限系统：撤销所有权限
```

**场景2：员工停用**
```javascript
// 1. 修改员工状态为 'suspended'
await employeeAPI.updateEmployeeStatus(id, 'suspended');

// 2. 处理关联数据
// - 暂停客户分配
// - 暂停任务分配
// - 保留数据访问权限（只读）

// 3. 可以重新激活
await employeeAPI.updateEmployeeStatus(id, 'active');
```

#### 部门删除时的员工处理

**方案1：阻止删除**
```javascript
// 检查部门下是否有员工
const employeeCount = await employeeAPI.getEmployees({
  department: departmentId
});

if (employeeCount > 0) {
  message.error(`该部门下还有 ${employeeCount} 个员工，无法删除`);
  return;
}
```

**方案2：转移员工**
```javascript
// 删除部门前，将员工转移到其他部门
Modal.confirm({
  title: '部门下有员工',
  content: '请先将员工转移到其他部门',
  onOk: () => {
    // 打开员工转移对话框
    showTransferDialog(departmentId);
  }
});
```

**方案3：设置为未分配**
```javascript
// 将部门下的员工设置为"未分配部门"
await employeeAPI.batchUpdate({
  department: null,
  where: { departmentId }
});
```

#### 与客户管理模块的关联

**客户分配**
```javascript
// 分配客户时，检查员工状态
const employee = await employeeAPI.getEmployee(employeeId);

if (employee.status !== 'active') {
  message.error('该员工已离职或停用，无法分配客户');
  return;
}

// 分配客户
await clientAPI.assignClient({
  clientId,
  employeeId,
  assignedBy: currentUser.id
});
```

**客户转移**
```javascript
// 员工离职时，自动转移客户
const handleEmployeeResign = async (employeeId) => {
  // 1. 获取该员工的所有客户
  const clients = await clientAPI.getClients({
    assignedTo: employeeId
  });

  // 2. 选择新的负责人
  const newEmployeeId = await selectNewEmployee();

  // 3. 批量转移客户
  await clientAPI.batchTransfer({
    clientIds: clients.map(c => c.id),
    fromEmployeeId: employeeId,
    toEmployeeId: newEmployeeId,
    reason: '原负责人离职'
  });

  // 4. 更新员工状态
  await employeeAPI.updateEmployeeStatus(employeeId, 'inactive');
};
```

/**
 * 数据同步服务
 * 处理员工模块与其他模块的数据一致性
 */
class SyncService {
  /**
   * 员工离职处理
   */
  async handleEmployeeResignation(employeeId) {
    try {
      // 1. 获取员工信息
      const employee = await employeeAPI.getEmployee(employeeId);

      // 2. 处理客户分配
      await this.transferEmployeeClients(employeeId);

      // 3. 处理任务分配
      await this.transferEmployeeTasks(employeeId);

      // 4. 撤销权限
      await this.revokeEmployeePermissions(employeeId);

      // 5. 更新员工状态
      await employeeAPI.updateEmployeeStatus(employeeId, 'inactive');

      // 6. 记录操作日志
      await this.logOperation({
        type: 'employee_resignation',
        employeeId,
        timestamp: new Date()
      });

      return { success: true };
    } catch (error) {
      console.error('员工离职处理失败:', error);
      throw error;
    }
  }

  /**
   * 转移员工客户
   */
  async transferEmployeeClients(employeeId) {
    // 获取该员工的所有客户
    const clients = await clientAPI.getClients({
      assignedTo: employeeId,
      status: 'active'
    });

    if (clients.length === 0) return;

    // 选择新的负责人（部门经理或指定人员）
    const newEmployeeId = await this.selectNewEmployee(employeeId);

    // 批量转移
    await clientAPI.batchTransfer({
      clientIds: clients.map(c => c.id),
      fromEmployeeId: employeeId,
      toEmployeeId: newEmployeeId,
      reason: '原负责人离职'
    });
  }

   * 转移员工任务

  }

  /**
   * 撤销员工权限

  }

   * 选择新的负责人

/**
 * 通知服务
 */
class NotificationService {
  /**
   * 员工创建通知
   */
  async notifyEmployeeCreated(employee, initialPassword) {
    // 发送邮件通知
    await this.sendEmail({
      to: employee.email,
      subject: '欢迎加入公司',
      template: 'employee_welcome',
      data: {
        name: employee.name,
        username: employee.username,
        password: initialPassword,
        loginUrl: window.location.origin + '/login'
      }
    });

    // 发送短信通知
    if (employee.phone) {
      await this.sendSMS({
        to: employee.phone,
        message: `欢迎加入公司！您的登录账号是：${employee.username}，初始密码已发送至邮箱。`
      });
    }
  }

  /**
   * 密码重置通知
   */
  async notifyPasswordReset(employee, newPassword) {
    await this.sendEmail({
      to: employee.email,
      subject: '密码已重置',
      template: 'password_reset',
      data: {
        name: employee.name,
        password: newPassword
      }
    });
  }

  /**
   * 员工离职通知
   */
  async notifyEmployeeResignation(employee) {
    // 通知部门经理
    const manager = await employeeAPI.getDepartmentManager(employee.department);
    if (manager) {
      await this.sendEmail({
        to: manager.email,
        subject: '员工离职通知',
        template: 'employee_resignation',
        data: {
          employeeName: employee.name,
          department: employee.department,
          resignDate: new Date()
        }
      });
    }
  }

   * 发送邮件
   */

  }

}
